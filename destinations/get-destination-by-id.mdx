---
title: 'Get Destination by Id'
api: 'POST https://api3.cloudsafaris.com/v1/get-destination-with-id'
---

### Header Parameters
<ParamField header="CloudSafaris-API-Version" type="string" default="v1">
  Currently the only acceptable value is `v1`.
</ParamField>

### Request Body Parameters

<ParamField body="id" type="number">
  The unique identifier for the destination.
</ParamField>

<RequestExample>
  ```json
  {
    "id": 1
  }
```
</RequestExample>

### Response Body Parameters

<ResponseField name="data" type="object">
<Expandable name="properties" defaultOpen={true}>
    <ResponseField name="id" type="number">
        The unique identified for the destination.
    </ResponseField>

    <ResponseField name="country" type="string">
        The country that the destination is in.
    </ResponseField>

    <ResponseField name="name" type="string">
        The name of the destination.
    </ResponseField>

    <ResponseField name="description" type="string">
        The description for the destination.
    </ResponseField>

    <ResponseField name="short_description" type="string">
        A short description for the destination.
    </ResponseField>

    <ResponseField name="geographic_size" type="number">
        If this is a region, geographic size will give the size of the area (km^2).
    </ResponseField>

    <ResponseField name="altitude" type="number">
        If the destination is at a certain altitude this will return the altitude in meters.
    </ResponseField>

    <ResponseField name="population" type="number">
        Population data for the destination.
    </ResponseField>

    <ResponseField name="best_time_to_visit" type="string">
        Information related to the best time to visit a destination.
    </ResponseField>

    <ResponseField name="latitude" type="number">
        The latitude of the destination.    
    </ResponseField>

    <ResponseField name="longitude" type="number">
        The longitude of a destination
    </ResponseField>

    <ResponseField name="images" type="string[]">
        The image URLs for a given destination.
    </ResponseField>

    <ResponseField name="destination_type" type="string">
        Takes on one of the following values:
        `"National Park", "Wildlife Conservancy", "International Airport", "Domestic Airport",
    "Airstrip", "Mountain", "City", "Village", "Beach", "Market", "Museums and Art Galleries", "Vineyards and Wineries", "Not Specified", 
    "Landmark", "Geographical feature", "Camp", "Region", "Peak", "Gate", "Lake"`    
    </ResponseField>

    <ResponseField name="faq" type="object[]">
        <Expandable name="properties" defaultOpen={true}>
            <ResponseField name="question" type="string">
                The question as per the FAQ provided.
            </ResponseField>
            <ResponseField name="answer" type="string">
                The answer to the applicable question in the FAQ.
            </ResponseField>
        </Expandable>    
    </ResponseField>

    <ResponseField name="extra_details" type="object[]">
        Extra details contain information about a destination like "Getting Around" or "Visas".
        <Expandable name="properties" defaultOpen={true}>
            <ResponseField name="type" type="string">
                The type of extra detail that is provided. Possible values listed below:
                `"Getting Around", "Safety & Vaccinations", "Not Specified", "Getting There", "Visas"`
            </ResponseField>
            <ResponseField name="title" type="string">
                The title for the extra detail.
            </ResponseField>
            <ResponseField name="description" type="string">
                The description for the extra detail.
            </ResponseField>
            <ResponseField name="url" type="string">
                A URL for the given extra detail. This field may be empty.
            </ResponseField>
        </Expandable>
    </ResponseField>

    <ResponseField name="wildlife" type="object[]">
        An array of data pertaining to the wildlife at a particular destination.
        <Expandable name="properties" defaultOpen={true}>
            <ResponseField name="name" type="string">
                Name of the wildlife (I.e. Lion).
            </ResponseField>
            <ResponseField name="description" type="string">
                Description for the wildlife.
            </ResponseField>
            <ResponseField name="images" type="string[]">
                The image URLs for the destination.
            </ResponseField>
            <ResponseField name="rarity" type="How rare the wildlife is">
                How rare the wildlife is. Possible values are listed below:
                `"Common", "Rare", "Occasional", "Not Present"`
            </ResponseField>
        </Expandable>
    </ResponseField>

</Expandable>
</ResponseField>
