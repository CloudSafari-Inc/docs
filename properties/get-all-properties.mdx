---
title: 'Get all Properties'
api: 'POST https://api3.cloudsafaris.com/v1/properties/get-properties'
---

### Header Parameters
<ParamField header="CloudSafaris-API-Version" type="string" default="v1">
  Currently the only acceptable value is `v1`.
</ParamField>

### Request Body Parameters

<ParamField body="limit" type="number" defaultValue={20}>
    A limit to the number of destinations to be returned.
</ParamField>
<ParamField body="offset" type="number" defaultValue={0}>
    The offset to start from to get the next set of destinations.
</ParamField>
<ParamField body="countries" type="string[]" defaultValue={0}>
    Returns back properties that are in a certain country. Possible values are `TZ` and `KE`.
</ParamField>

<RequestExample>
  ```json
  {
    "limit": 1,
    "offset": 0,
    "destinations": [1, 2, 3],
    "luxury_level": 1,
    "types": ["Lodge", "Mobile Camp", "Hotel"], 
    "Amenities": ["Hair Dryer", "Wifi", "Lounge & Bar"], 
  }
```
</RequestExample>

### Response Body Parameters

<ResponseField name="data" type="object[]">
    <Expandable name="properties" defaultOpen={true}>
        <ResponseField name="name" type="string">
            Name of the property
        </ResponseField>
        <ResponseField name="latitude" type="number">
            Latitude for the property.
        </ResponseField>
        <ResponseField name="longitude" type="number">
            Longitude for the property.
        </ResponseField>
        <ResponseField name="social_links" type="string[]">
            Social links for the property. This is the full URL for the social link such as `https://instagram.com/cloudsafaris`.
        </ResponseField>
        <ResponseField name="note" type="string">
            A note pertaining to the property details.
        </ResponseField>
        <ResponseField name="images" type="string[]">
            An array of strings with the image URLs that can be directly used to access images for the properties.
        </ResponseField>
        <ResponseField name="amenities" type="object[]">
            Amenities that the property has.
            <Expandable name="properties" defaultOpen={true}>
                <ResponseField name="name" type="string">
                    Name of the amenity.
                </ResponseField>
                <ResponseField name="description" type="string">
                    Description for the amenity.
                </ResponseField>
                <ResponseField name="icon_url" type="string">
                    An icon that represents the amenity.
                </ResponseField>
            </Expandable>
        </ResponseField>
        <ResponseField name="highlights" type="string[]">
            An array of strings for the highlights of the property.
        </ResponseField>
        <ResponseField name="getting_there" type="object[]">
            <Expandable name="properties" type="object">
                <ResponseField name="type" type="string">
                    The type corresponding on how to get to the property (I.e. By flight, by vehicle).
                </ResponseField>
                <ResponseField name="description" type="string">
                    The details on how to get to the property by the given type.
                </ResponseField>
            </Expandable>
        </ResponseField>
        <ResponseField name="room_description" type="string">
            A description on the rooms in the property.
        </ResponseField>
    </Expandable>
</ResponseField>
